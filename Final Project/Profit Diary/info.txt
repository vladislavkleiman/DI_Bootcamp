CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    username VARCHAR(50) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL
);



CREATE TABLE tradestransaction (
    id_trades SERIAL PRIMARY KEY,
    user_id INTEGER,
    DateTrades DATE,
    Currency TEXT,
    Side TEXT,
    Symbol TEXT,
    Qty INTEGER,
    Price NUMERIC,
    ExecTime TIME,
    GrossProceeds NUMERIC,
    NetProceeds NUMERIC
);

CREATE TABLE statisticoftradesforday (
    id SERIAL PRIMARY KEY,
    date_trade DATE,
    user_id VARCHAR,
    profitloss NUMERIC,
    total_winners INTEGER,
    total_losers INTEGER,
    total_trades INTEGER,
    avg_return NUMERIC,
    total_long INTEGER,
    total_shorts INTEGER,
    win_rate NUMERIC
);




CREATE TABLE trades (
    id_trades SERIAL PRIMARY KEY,
    user_id VARCHAR,
    trade_date DATE,
    exectime TIME,
    stock_ticker TEXT,
    trade_type TEXT,
    profit_loss VARCHAR
);
